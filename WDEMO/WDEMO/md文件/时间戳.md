 #时间戳
 iOS是秒
 java一般都是毫秒
```
NSDate *data = [NSDate date];
NSTimeInterval time = [data timeIntervalSince1970];
```

```
//字符串转时间戳
//datenow为当前时间
NSDate *datenow = [NSDate date];
NSString *timeSp = [NSString stringWithFormat:@"%d", (long)[datenow timeIntervalSince1970]]; 
//时间戳的值
NSLog(@"timeSp:%@",timeSp);
```
