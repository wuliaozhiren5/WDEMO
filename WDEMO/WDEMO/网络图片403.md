#网络图片403
SDWebImage

```
- (void)rr_downloadImageWithURLString:(nullable NSString *)url
                     placeholderImage:(nullable UIImage *)placeholder {
    self.rr_delayImageURLString = nil;
    if (!canUseObj(url)) {
        self.image = placeholder;
        return;
    }
    [self sd_setImageWithURL:URL([RRMJTool thumbnailUrlString:url withImageWidth:self.width]) placeholderImage:placeholder completed:^(UIImage * _Nullable image, NSError * _Nullable error, SDImageCacheType cacheType, NSURL * _Nullable imageURL) {
        if (image) {
            [self setImage:image];
        } else {
            if (error) {
                NSNumber *codeNumber = error.userInfo[@"SDWebImageErrorDownloadStatusCodeKey"];
                NSInteger code = codeNumber.integerValue;
                NSLog(@"%zi",code); 
                if(code == 403){
                    [self setImage:IMAGENAME(@"forbid")];
                }
            }
        }
    }];
}
```
